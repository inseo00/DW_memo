1) 사원명, 부서번호, 부서이름, 부서 근무지 조회
select 
 e.ename as '사원명',
 e.DEPTNO as '부서번호',
 d.dname as '부서이름',
 d.loc as '부서근무지'
from 
emp e
join 
dept d 
on 
e.DEPTNO = d.DEPTNO;

-----------------------------------------------------------

2) 부서이름별 급여합계, 부서이름을 조회
select 
sum(sal) as '급여합계',
d.DNAME as '부서이름'
from 
emp e
join 
dept d 
on 
e.DEPTNO = d.DEPTNO  
group by e.DEPTNO;
------------------------------------------------------------
3)부서 근무지가 new york이고, 직책이 manager인 사원의 이름, 급여 조회
select 
 e.ENAME as '사원이름' ,
 e.SAL as '급여'
from 
emp e
join 
dept d 
on 
e.DEPTNO  = d.DEPTNO 
where
d.LOC = 'new york'
and
e.job = 'manager';

4) 1983년 이후 입사한 사원의 보서수가 null이면 100을 주고 
  사원의 이름, 부서 이름, 직업을 조회
 select 
 e.ENAME as '사원이름',
 d.DNAME as '부서이름',
 e.JOB as '직업',
 date_format(HIREDATE, '%Y'),
 ifnull(comm, 100) as '보너스'

emp e

dept d

e.DEPTNO  = d.DEPTNO
